// 右侧搜索功能
function preSearchSearch(params) {
  // 解构参数
  const { type, value } = params
  console.log(typeof (value));
  console.log(value);
  // 对不同字段进行过滤搜索
  if (Array.isArray(value)) {
  totalData.value = totalData.value.filter(item => {
    return value.some(valueItem => {
      return valueItem.some(valueItemItem => {
        return item[type] && item[type].includes(valueItemItem)
      });
    });
  });
  } else {
    totalData.value = totalData.value.filter(item => item[type].includes(value))
  }
}
